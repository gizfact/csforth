/* Пример программы */

: Test
	if "true" else "false" end
;

: Divide
	1.0 8.1 /d .d
;
	
: Fact
  dup 1.0 >d if
    dup 1.0 -d Fact *d
  else
    drop 1.0
  end
;

class string
	: .s
		"Hello from class: " swap +s .s
	;
;

: Goal
	1.5 ->i Test Divide .s 100.0 Fact .d
	"Hello, world!" dup " world" dup -rot s.pos swap rot .s .s .
	-1 dup asr swap asl 2 "Хрень: " "{0:X}, {1:X}" +s format string .s
; 

Goal
